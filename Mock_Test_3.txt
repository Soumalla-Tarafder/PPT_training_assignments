'''
Implement a stack using a list in Python. Include the necessary 
methods such as push, pop, and isEmpty.

'''
class stack:
    def __init__(self):
        self.l=[]
    
    def push(self,val):
        self.l.append(val)
        print(self.l)
        
    def is_empty(self):
        if len(self.l) == 0:
            print("True")
            return True
        else:
            print("False")
            return False
    def pop(self):
        
        if len(self.l)!=0:
            self.l.pop()
            print(self.l)
        else:
            print("No Data To Pop")
            
a=stack()

a.is_empty()

a.push(5)
a.push(7)

a.pop()

a.is_empty()
a.pop()
a.pop()
